cat <<EOF > check.sh
#!/bin/bash

COMPLETE=0
TODO=0
ERROR=0

cd /gluster/wgp/wgp/sequencing/colleagues_data/Andrew_fry/DDD

for EGAF in \`cut -f 4 DDD_mcd_fileset_reference.tsv | sed 1d\`

        OK=\`grep "Download complete" \$EGAF.log | wc -l\`
        #echo "\$EGAF \$OK"
        if [ \$OK -gt 0 ]; then
            COMPLETE=\$((COMPLETE+1)) 
            #echo "OK: \$EGAF \$OK"  
        else 
            ERROR=\$((ERROR+1))   
            #echo "ERROR: \$EGAF \$OK"  
        fi
        
    else
        TODO=\$((TODO+1))
    fi


done

echo "Complete: \$COMPLETE"
echo "Failed: \$ERROR"
echo "To-Do: \$TODO"

EOF
